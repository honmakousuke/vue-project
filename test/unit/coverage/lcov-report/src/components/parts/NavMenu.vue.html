      items: [

        { title: 'SKILLS', path: '/skills' },

      activeItem: null,
      ActiveBtn: false

  methods: {
    onActive (item) {
      if (this.activeItem === item) {
        this.activeItem = null

        this.activeItem = item
        this.ActiveBtn = false
      }
    }
  }
}
